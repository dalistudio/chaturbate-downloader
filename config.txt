#设置useragent
--user-agent  "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/65.0.3325.181 Safari/537.36"
#设置输出位置
-o '%(id)s/%(title)s.%(ext)s'

#--no-part
#避免文件中的空格或者&
--restrict-filenames
